Actions
draw = () => {
    card = library.pop()
    playerHand + card
}
discard = (...cards) => {
    for each card in cards
        playerhand - card
        graveyard + card
}
scry = (x) => {
    for i in range x
        view += library[i]
}
shuffle = () => {
    randomize library
}
proliferate = (...counters) => {
    for each counter in counters:
        counter.numCounter ++
}
sacrifice = (...cards) => {
    for each card in cards:
        battlefield - card
        graveyard + card
}
exile = (...cards) => {
    for each card in cards:
        battlefield - card
        exile + card
}
enchant = (...cards) => {
    for each card in cards:
        card.enchantments + card
}
destroy = (...cards) => {
    for each card in cards:
        battlefield - card
        graveyard + card
}
goad = (...cards) => {
    for each card in cards:
    card.owner.attackers + card
}
